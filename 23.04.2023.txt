#include <iostream>
using namespace std;

struct animal {
    char* name{};
    int age{};
    char gender{};
};

struct Zoo {
    char* name{};
    int num1{};
    int num2{};
    animal animals[100]{};


    void add(animal num3) {
        if (num1 < num2) {
            animals[num1] = num3;
            num1++;
            cout << "Animal added to zoo" << endl;
        } else {
            cout << "Zoo is full" << endl;
        }
    }

    void remove(int num4) {
        if (num4 >= 0 && num4 < num1) {
            for (int i = num4; i < num1 - 1; i++) {
                animals[i] = animals[i + 1];
            }
            num1--;
            cout << "Animal removed from zoo" << endl;

        } else {
            cout << "Invalid" << endl;
        }
    }

    void edit(int num4, animal num3) {
        if (num4 >= 0 && num4 < num1) {
            animals[num4] = num3;
            cout << "Animal edited" << endl;
        }
        else {
            cout << "Invalid" << endl;
        }
    }

    void fun() {
        cout << "Animals in zoo:" << endl;
        for (int i = 0; i < num1; i++) {
            cout << i + 1 << ".Name: " << animals[i].name << ", Age: " << animals[i].age << ", Gender: " << animals[i].gender << endl;
        }
    }


};

int main() {
    Zoo zoo2{};

    zoo2.name = "Ali's Zoo";
    zoo2.num2 = 100;

    animal lion {"Simba", 10, 'M'};
    zoo2.add(lion);

    animal elephant {"Veronika", 7, 'W'};
    zoo2.add(elephant);

    animal horse {"Tom", 5, 'M'};
    zoo2.add(horse);


    zoo2.fun();

    animal lion2 {"Emma", 4, 'W'};
    zoo2.edit(0, lion2);
    zoo2.fun();

    zoo2.remove(1);

    zoo2.fun();

    return 0;
}
